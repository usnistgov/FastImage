116 void
117 AABBTree<ObjectT>::preprocess(std::vector<ObjectT> *objects) {
118 
119  aabbTree.clear();
120 
121  sortedObjects = objects;
122 
123  if (objects->empty()) {
124  return;
125  }
126 
127  aabbTree.resize(1);
128 
129  aabbTree.at(0).objectsBegin = 0;
130  aabbTree.at(0).objectsEnd = sortedObjects->size();
131 
132  aabbTree_size_type nodesBegin = 0;
133  aabbTree_size_type nodesEnd = 1;
134 
135  int splitDim = 0;
136 
137  bool splitOccurred = true;
138 
139  while (splitOccurred) {
140  splitOccurred = false;
141 
142  for (aabbTree_size_type nodeIndex = nodesBegin;
143  nodeIndex < nodesEnd;
144  ++nodeIndex) {
145 
146  computeAABBBounds(nodeIndex);
147 
148  objects_size_type nodeSize =
149  aabbTree.at(nodeIndex).objectsEnd -
150  aabbTree.at(nodeIndex).objectsBegin;
151 
152  if (nodeSize > maxNodeSize) {
153  if (!splitOccurred) {
154  aabbTree_size_type newAABBTreeSize = aabbTree.size() * 2 + 1;
155 
156  aabbTree.resize(newAABBTreeSize);
157  }
158 
159  splitAABBNode(nodeIndex, splitDim);
160 
161  splitOccurred = true;
162  }
163  }
164 
165  nodesBegin = nodesEnd;
166  nodesEnd = nodesEnd * 2 + 1;
167 
168  ++splitDim;
169 
170  if (splitDim == 2) {
171  splitDim = 0;
172  }
173  }
174 }

178 AABBTree<ObjectT>::computeAABBBounds(aabbTree_size_type nodeIndex) {
179  for (int dim = 0; dim < 2; ++dim) {
180  aabbTree.at(nodeIndex).aabb.setMaxCoord(
181  dim, std::numeric_limits<double>::lowest());
182  }
183 
184  for (int dim = 0; dim < 2; ++dim) {
185  aabbTree.at(nodeIndex).aabb.setMinCoord(
186  dim, std::numeric_limits<double>::max());
187  }
188 
189  for (objects_size_type objectIndex =
190  aabbTree.at(nodeIndex).objectsBegin;
191  objectIndex < aabbTree.at(nodeIndex).objectsEnd;
192  ++objectIndex) {
193 
194  for (int dim = 0; dim < 2; ++dim) {
195  aabbTree.at(nodeIndex).aabb.
196  setMaxCoord(dim,
197  std::max(
198  aabbTree.at(nodeIndex).aabb.getMaxCoord(dim),
199  sortedObjects->at(objectIndex).getMaxCoord(dim)));
200  }
201 
202  for (int dim = 0; dim < 2; ++dim) {
203  aabbTree.at(nodeIndex).aabb.
204  setMinCoord(dim,
205  std::min(
206  aabbTree.at(nodeIndex).aabb.getMinCoord(dim),
207  sortedObjects->at(objectIndex).getMinCoord(dim)));
208  }
209  }

214 AABBTree<ObjectT>::splitAABBNode(aabbTree_size_type nodeIndex,
215  int splitDim) {
216 
217  class CoordComp {
218  public:
219  explicit CoordComp(int dim) : dim(dim) {}
220 
221  bool operator()(ObjectT i, ObjectT j) {
222  double iMidCoordX2 = i.getMaxCoord(dim) + i.getMinCoord(dim);
223  double jMidCoordX2 = j.getMaxCoord(dim) + j.getMinCoord(dim);
224 
225  return (iMidCoordX2 < jMidCoordX2);
226  }
227 
228  private:
229  int dim;
230  };
231 
232  CoordComp coordComp(splitDim);
233 
234  std::sort(sortedObjects->begin() + aabbTree.at(nodeIndex).objectsBegin,
235  sortedObjects->begin() + aabbTree.at(nodeIndex).objectsEnd,
236  coordComp);
237 
238  objects_size_type nodeSize =
239  aabbTree.at(nodeIndex).objectsEnd -
240  aabbTree.at(nodeIndex).objectsBegin;
241 
242  objects_size_type objectsMiddle =
243  aabbTree.at(nodeIndex).objectsBegin +
244  nodeSize / 2;
245 
246  aabbTree_size_type lowChildIndex = nodeIndex * 2 + 1;
247  aabbTree_size_type highChildIndex = nodeIndex * 2 + 2;
248 
249  aabbTree.at(lowChildIndex).objectsBegin =
250  aabbTree.at(nodeIndex).objectsBegin;
251 
252  aabbTree.at(lowChildIndex).objectsEnd =
253  objectsMiddle;
254 
255  aabbTree.at(highChildIndex).objectsBegin =
256  objectsMiddle;
257 
258  aabbTree.at(highChildIndex).objectsEnd =
259  aabbTree.at(nodeIndex).objectsEnd;
260 
261  aabbTree.at(nodeIndex).objectsBegin = 0;
262  aabbTree.at(nodeIndex).objectsEnd = 0;
263 }

267 template<class ObjectT>
268 std::vector<ObjectT *> AABBTree<ObjectT>::objectsContain(
269  Vector2<double> const &queryPoint) const {
270 
271  std::vector<ObjectT *> vObj;
272  if (aabbTree.empty()) {
273  return vObj;
274  }
275 
276  std::vector<aabbTree_size_type> nodesToCheck;
277 
278  nodesToCheck.emplace_back(0);
279 
280  while (!nodesToCheck.empty()) {
281  aabbTree_size_type nodeIndex = nodesToCheck.back();
282 
283  nodesToCheck.pop_back();
284 
285  objects_size_type objectsBegin = aabbTree.at(nodeIndex).objectsBegin;
286  objects_size_type objectsEnd = aabbTree.at(nodeIndex).objectsEnd;
287 
288  if (objectsBegin == objectsEnd) {
289  aabbTree_size_type lowChildIndex = nodeIndex * 2 + 1;
290  aabbTree_size_type highChildIndex = nodeIndex * 2 + 2;
291 
292  if (aabbTree.at(highChildIndex).aabb.contains(queryPoint)) {
293  nodesToCheck.emplace_back(highChildIndex);
294  }
295 
296  if (aabbTree.at(lowChildIndex).aabb.contains(queryPoint)) {
297  nodesToCheck.emplace_back(lowChildIndex);
298  }
299  } else {
300  // leaf node
301  for (objects_size_type objectIndex = objectsBegin;
302  objectIndex < objectsEnd;
303  ++objectIndex) {
304  auto feature = sortedObjects->at(objectIndex);
305  feature.getId();
306  if (sortedObjects->at(objectIndex).contains(queryPoint)) {
307  vObj.push_back(&sortedObjects->at(objectIndex));
308  }
309  }
310  }
311  }
312  return vObj;
313 }
314 }
